--------------------------------------------------------------------

--------------------------------------------------------------------

--------------------------------------------------------------------
alias c='clear'

Show Date and Time
--------------------------------------------------------------------
alias now='date +%A.%B.%d,%Y'

--------------------------------------------------------------------

--------------------------------------------------------------------

--------------------------------------------------------------------
